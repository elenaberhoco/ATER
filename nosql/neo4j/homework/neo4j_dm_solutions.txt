MATCH (m:Match)

  WHEN "First Stage" THEN "First stage"
  WHEN "Match for 3rd place" THEN "Match for third place"
  WHEN "Quarter Finals" THEN "Quarter finals"
  WHEN "Quarter-final" THEN "Quarter finals"
  WHEN "Quarter-finals" THEN "Quarter finals"
  WHEN "Round of 16" THEN "Round of sixteen"
  WHEN "Semi-final" THEN "Semi finals"
  WHEN "Semi-finals" THEN "Semi finals"
  WHEN "Semifinals" THEN "Semi finals" 
  WHEN "Group Matches" THEN "Group matches"
  ELSE m.stage
